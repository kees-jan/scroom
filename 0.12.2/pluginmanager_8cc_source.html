34 bool PluginManager::doWork()
35 {
36  bool retval = true;
37 
38  gdk_threads_enter();
39  switch(state)
40  {
41  case FINDING_DIRECTORIES:
42  {
43  setStatusBarMessage("Locating plugin directories");
44  char* path = getenv(SCROOM_PLUGIN_DIRS.c_str());
45  dirs.clear();
46 
47  if(!devMode)
48  {
49 #ifdef _WIN32
50  // We want to keep everything portable on windows so we look for the plugin folder in the same directory as the .exe
51  std::string plugin_path = (boost::dll::program_location().parent_path() / "plugins").generic_string();
52  dirs.push_back(plugin_path);
53 #else
54  dirs.emplace_back(PLUGIN_DIR);
55 #endif
56  }
57 
58  if(path != nullptr)
59  {
60  printf("%s = %s\n", SCROOM_PLUGIN_DIRS.c_str(), path);
61 
62  for(char* i = path; *i != '\0'; i++)
63  {
64 
65 // Windows uses semicolons for delimiting environment variables, Linux uses colons
66 #ifdef _WIN32
67  const char envDelim = ';';
68 #else
69  const char envDelim = ':';
70 #endif
71  if(*i != envDelim)
72  {
73  continue;
74  }
75 
76  *i = '\0';
77  dirs.emplace_back(path);
78  path = i + 1;
79  }
80 
81  dirs.emplace_back(path);
82  }
83 
84  currentDir = dirs.begin();
85  files.clear();
86  state = SCANNING_DIRECTORIES;
87  break;
88  }
89 
90  case SCANNING_DIRECTORIES:
91  {
92  setStatusBarMessage("Scanning plugin directories");
93  if(currentDir == dirs.end())
94  {
95  state = LOADING_FILES;
96  currentFile = files.begin();
97  break;
98  }
99 
100  printf("Scanning directory: %s\n", currentDir->c_str());
101  const char* folder = currentDir->c_str();
102  namespace fs = boost::filesystem;
103  boost::system::error_code ec;
104 
105  if(!fs::is_directory(folder, ec))
106  {
107  printf("Can't open directory...\n");
108  currentDir++;
109  break;
110  }
111 
112  for(const auto& entry: fs::directory_iterator(folder))
113  {
114  if(fs::is_regular_file(entry) || fs::is_symlink(entry) || fs::is_other(entry))
115  {
116  files.push_back(entry.path().generic_string());
117  }
118  }
119 
120  currentDir++;
121  break;
122  }
123 
124  case LOADING_FILES:
125  {
126  setStatusBarMessage("Loading Plugins");
127  if(currentFile == files.end())
128  {
129  state = DONE;
130  break;
131  }
132 
133 #ifdef _WIN32
134  // Only read .dll files
135  if(currentFile->compare(currentFile->size() - 4, 4, ".dll") == 0)
136  {
137 #else
138  // Only read .so files
139  if(currentFile->compare(currentFile->size() - 3, 3, ".so") == 0)
140  {
141 #endif
142  printf("Reading file: %s\n", currentFile->c_str());
143  GModule* plugin = g_module_open(currentFile->c_str(), static_cast<GModuleFlags>(0));
144  if(plugin)
145  {
146  // Need to pass a gpointer to g_module_symbol. If I pass a
147  // PluginFunc, glib 2.16.6/gcc 4.2.4 will complain about
148  // type-punned pointers.
149  gpointer pgpi;
150  if(g_module_symbol(plugin, "getPluginInformation", &pgpi))
151  {
152  auto gpi = reinterpret_cast<PluginFunc>(pgpi);
153  if(gpi)
154  {
155  PluginInformationInterface::Ptr pi = (*gpi)();
156  if(pi)
157  {
158  if(pi->pluginApiVersion == PLUGIN_API_VERSION)
159  {
160  pluginInformationList.emplace_back(plugin, pi);
161  pi->registerCapabilities(shared_from_this<PluginManager>());
162  plugin = nullptr;
163  gpi = nullptr;
164  pi.reset();
165  }
166  else
167  {
168  printf("Plugin has incorrect API version %d, instead of %d\n", pi->pluginApiVersion, PLUGIN_API_VERSION);
169  }
170  }
171  else
172  {
173  printf("GetPluginInformation returned NULL\n");
174  }
175  }
176  else
177  {
178  printf("Can't find the getPluginInterface function: %s\n", g_module_error());
179  }
180  }
181  else
182  {
183  printf("Can't lookup symbols: %s\n", g_module_error());
184  }
185 
186  if(plugin)
187  {
188  g_module_close(plugin);
189  }
190  }
191  else
192  {
193  printf("Something went wrong: %s\n", g_module_error());
194  }
195  }
196  }
197  currentFile++;
198  break;
199  case DONE:
200  {
201  setStatusBarMessage("Done loading plugins");
202  on_done_loading_plugins();
203  retval = false;
204  break;
205  }
206  }
207  gdk_threads_leave();
208  return retval;
209 }
