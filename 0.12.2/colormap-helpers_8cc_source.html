104 void MonochromeColormapHelper::regenerateColormap()
105 {
106  Color w = whitish;
107  Color b = blackish;
108  if(transparentBackground)
109  {
110  if(inverted)
111  {
112  b.setAlpha(0);
113  }
114  else
115  {
116  w.setAlpha(0);
117  }
118  }
119 
120  for(int i = 0; i < numberOfColors; i++)
121  {
122  colormap->colors[i] = mix(w, b, 1.0 * i / (numberOfColors - 1));
123  }
124 }
