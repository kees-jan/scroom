70 void TransparentOverlayPresentation::setOptimalColor(PresentationInterface::Ptr const& p)
71 {
72  Colormappable::Ptr c = boost::dynamic_pointer_cast<Colormappable>(p);
73  if(c)
74  {
75  std::map<Color, int, ColorComparer> currentColors;
76  for(PresentationInterface::Ptr const& child: children)
77  {
78  Colormappable::Ptr cChild = boost::dynamic_pointer_cast<Colormappable>(child);
79  if(cChild && child->isPropertyDefined(MONOCHROME_COLORMAPPABLE_PROPERTY_NAME))
80  {
81  currentColors[cChild->getMonochromeColor()]++;
82  }
83  }
84 
85  Color minimumColor = c->getMonochromeColor();
86  int minimumColorValue = currentColors[minimumColor];
87 
88  for(Color const& color: colors)
89  {
90  if(currentColors[color] < minimumColorValue)
91  {
92  minimumColor = color;
93  minimumColorValue = currentColors[color];
94  }
95  }
96 
97  c->setMonochromeColor(minimumColor);
98  }
99 }
