138 void View::redraw(cairo_t* cr)
139 {
140  if(presentation)
141  {
142  GdkRectangle rect;
143  rect.x = x;
144  rect.y = y;
145  if(zoom >= 0)
146  {
147  // Zooming in. Smallest step is 1 presentation pixel, which is more than one window-pixel
148  int pixelSize = 1 << zoom;
149  rect.width = (drawingAreaWidth + pixelSize * aspectRatio.x - 1) / (pixelSize * aspectRatio.x);
150  rect.height = (drawingAreaHeight + pixelSize * aspectRatio.y - 1) / (pixelSize * aspectRatio.y);
151  }
152  else
153  {
154  // Zooming out. Smallest step is 1 window-pixel, which is more than one presentation-pixel
155  int pixelSize = 1 << (-zoom);
156  rect.width = drawingAreaWidth * (pixelSize / aspectRatio.x);
157  rect.height = drawingAreaHeight * (pixelSize / aspectRatio.y);
158 
159  // Round to whole pixels
160  rect.x = floor(rect.x / (pixelSize / aspectRatio.x)) * (pixelSize / aspectRatio.x);
161  rect.y = floor(rect.y / (pixelSize / aspectRatio.y)) * (pixelSize / aspectRatio.y);
162  }
163 
164  presentation->redraw(shared_from_this<View>(), cr, rect, zoom);
165 
166  for(const auto& renderer: postRenderers)
167  {
168  renderer->render(shared_from_this<View>(), cr, rect, zoom);
169  }
170  }
171  else
172  {
173  // A logo here would be nice...
174 
175  // char buffer[] = "View says \"Hi\"";
176  //
177  // cairo_move_to(cr, 50, 50);
178  // cairo_show_text(cr, buffer);
179  }
180 }

191 void View::setPresentation(PresentationInterface::Ptr presentation_)
192 {
193  View::Ptr me = shared_from_this<View>();
194 
195  if(presentation)
196  {
197  presentation->close(me);
198  presentation.reset();
199  }
200 
201  presentation = std::move(presentation_);
202 
203  if(presentation)
204  {
205  presentation->open(me);
206  presentationRect = presentation->getRect();
207  aspectRatio = presentation->getAspectRatio();
208  std::string s = presentation->getTitle();
209  if(s.length())
210  {
211  s = "Scroom - " + s;
212  }
213  else
214  {
215  s = "Scroom";
216  }
217  gtk_window_set_title(window, s.c_str());
218  }
219 
220  zoom = 0;
221  const int pixelSize = 1 << zoom;
222  x = -drawingAreaWidth / (pixelSize * aspectRatio.x) / 2;
223  y = -drawingAreaHeight / (pixelSize * aspectRatio.y) / 2;
224 
225  updateZoom();
226  updateScrollbars();
227  updateTextbox();
228  invalidate();
229 }

231 void View::updateScrollbar(GtkAdjustment* adj,
232  int zoom_,
233  double value,
234  double presentationStart,
235  double presentationSize,
236  double windowSize)
237 {
238  if(zoom_ >= 0)
239  {
240  // Zooming in. Smallest step is 1 presentation pixel, which is more than one window-pixel
241  int pixelSize = 1 << zoom_;
242  presentationStart -= windowSize / pixelSize / 2;
243  presentationSize += windowSize / pixelSize;
244 
245  gtk_adjustment_configure(adj,
246  value,
247  presentationStart,
248  presentationStart + presentationSize,
249  1,
250  3 * windowSize / pixelSize / 4,
251  windowSize / pixelSize);
252  }
253  else
254  {
255  // Zooming out. Smallest step is 1 window-pixel, which is more than one presentation-pixel
256  int pixelSize = 1 << (-zoom_);
257  presentationStart -= windowSize * pixelSize / 2;
258  presentationSize += windowSize * pixelSize;
259 
260  gtk_adjustment_configure(adj,
261  value,
262  presentationStart,
263  presentationStart + presentationSize,
264  pixelSize,
265  3 * windowSize * pixelSize / 4,
266  windowSize * pixelSize);
267  }

324 void View::updateZoom()
325 {
326  if(presentation)
327  {
328  int presentationHeight = presentationRect.height();
329  int presentationWidth = presentationRect.width();
330  int minZoom = 0;
331 
332  while(presentationHeight > drawingAreaHeight / 2 || presentationWidth > drawingAreaWidth / 2)
333  {
334  presentationHeight >>= 1;
335  presentationWidth >>= 1;
336  minZoom--;
337  }
338 
339  gtk_widget_set_sensitive(GTK_WIDGET(zoomBox), true);
340 
341  int zMax = MaxZoom - minZoom;
342  zMax = std::max(zMax, 1 + MaxZoom - zoom);
343  zMax = std::min<size_t>(zMax, sizeof(zoomfactor) / sizeof(zoomfactor[0]));
344 
345  gtk_list_store_clear(zoomItems);
346  for(int z = 0; z < zMax; z++)
347  {
348  GtkTreeIter iter;
349  gtk_list_store_insert_with_values(zoomItems, &iter, z, COLUMN_TEXT, zoomfactor[z], COLUMN_ZOOM, MaxZoom - z, -1);
350 
351  if(zoom == MaxZoom - z)
352  {
353  gtk_combo_box_set_active_iter(zoomBox, &iter);
354  }
355  }
356  }
357  else
358  {
359  gtk_widget_set_sensitive(GTK_WIDGET(zoomBox), false);
360  gtk_list_store_clear(zoomItems);
361  GtkTreeIter iter;
362  gtk_list_store_insert_with_values(zoomItems, &iter, 0, COLUMN_TEXT, zoomfactor[0], COLUMN_ZOOM, MaxZoom, -1);
363  }
364 }

486 {
487  if(event->direction == GDK_SCROLL_UP || event->direction == GDK_SCROLL_DOWN)
488  {
489  int newZoom = zoom + ((event->direction == GDK_SCROLL_UP) ? 1 : -1);
490  newZoom = std::min(MaxZoom, newZoom);
491 
492  GtkTreeIter iter;
493  for(bool valid = gtk_tree_model_get_iter_first(GTK_TREE_MODEL(zoomItems), &iter); valid;
494  valid = gtk_tree_model_iter_next(GTK_TREE_MODEL(zoomItems), &iter))
495  {
496  GValue value = G_VALUE_INIT;
497  gtk_tree_model_get_value(GTK_TREE_MODEL(zoomItems), &iter, COLUMN_ZOOM, &value);
498  int foundZoom = g_value_get_int(&value);
499 
500  if(foundZoom == newZoom)
501  {
502  on_zoombox_changed(newZoom, event->x, event->y);
503  gtk_combo_box_set_active_iter(zoomBox, &iter);

505  }

668  if(std::abs(event->x - cachedPoint.x) >= pixelSize * aspectRatio.x)
669  {
670  int delta = (event->x - cachedPoint.x) / (pixelSize * aspectRatio.x);
671  cachedPoint.x += delta * pixelSize * aspectRatio.x;
672  newx -= delta;
673  }
674  if(std::abs(event->y - cachedPoint.y) >= pixelSize * aspectRatio.y)
675  {
676  int delta = (event->y - cachedPoint.y) / (pixelSize * aspectRatio.y);
677  cachedPoint.y += delta * pixelSize * aspectRatio.y;
678  newy -= delta;
679  }
680  }
681  else
682  {
683  const int pixelSize = 1 << -zoom;
684  newx -= (event->x - cachedPoint.x) * (pixelSize / aspectRatio.x);
685  newy -= (event->y - cachedPoint.y) * (pixelSize / aspectRatio.y);
686  cachedPoint = eventToPoint(event);
687  }
688 
689  updateXY(newx, newy, OTHER);
690  }
691  else if((event->state & GDK_BUTTON3_MASK) && selection)
692  {
693  selection->end = windowPointToPresentationPoint(eventToPoint(event));
694  for(const auto& listener: selectionListeners)
695  {
696  listener->onSelectionUpdate(selection, shared_from_this<ViewInterface>());
697  }
698  invalidate();
699  }

799 {
800  GdkPoint result = {0, 0};
801 
802  if(zoom >= 0)
803  {
804  const int pixelSize = 1 << zoom;
805  result.x = x + (wp.x + (pixelSize * aspectRatio.x) / 2) / (pixelSize * aspectRatio.x); // Round to make measurements snap
806  result.y = y + (wp.y + (pixelSize * aspectRatio.y) / 2) / (pixelSize * aspectRatio.y); // in the expected direction
807  }
808  else
809  {
810  const int pixelSize = 1 << -zoom;
811  result.x = x + wp.x * (pixelSize / aspectRatio.x);
812  result.y = y + wp.y * (pixelSize / aspectRatio.y);
813  }
