72 void TiledBitmapViewData::resetNeededTiles()
73 {
74  boost::unique_lock<boost::mutex> lock(mut);
75 
76  // If we just cleared out oldStuff, old tiles would be unloaded, and
77  // we might still need them. So create a backup to throw away later
78  std::list<boost::shared_ptr<void>> oldStuff;
79  oldStuff.swap(stuff);
80  std::list<boost::shared_ptr<void>> oldVolatileStuff;
81  oldVolatileStuff.swap(volatileStuff);
82 
83  lock.unlock();
84  // The stuff list contains both registrations and references to needed tiles.
85  // Registering an observer can result in tileLoaded() being called immediately
86  // if the tile was already loaded. Hence, we cannot hold the lock while registering
87  // observers.
88 
89  std::list<boost::shared_ptr<void>> newStuff;
90  // If we don't hold the lock, we cannot add to the stuff list directly. Hence,
91  // temporarily add registrations to the newStuff list, and add the newStuff to
92  // stuff later.
93 
94  for(int i = imin; i < imax; i++)
95  {
96  for(int j = jmin; j < jmax; j++)
97  {
98  CompressedTile::Ptr tile = layer->getTile(i, j);
99 
100  TileViewState::Ptr tileViewState = tile->getViewState(viewInterface);
101  tileViewState->setViewData(shared_from_this<TiledBitmapViewData>());
102  tileViewState->setZoom(layerOperations, zoom);
103  newStuff.emplace_back(tileViewState);
104  newStuff.emplace_back(tileViewState->registerObserver(shared_from_this<TiledBitmapViewData>()));
105  }
106  }
107 
108  // At this point, everything we need is either in the stuff list, or the newStuff list.
109  // Hence, this is an excellent time to clear the oldStuff list. We cannot clear the
110  // oldStuff list while holding the lock, because that would result in deadlock (see
111  // ticket #35)
112  oldStuff.clear();
113  oldVolatileStuff.clear();
114 
115  // Re-acquire the lock
116  lock.lock();
117  stuff.splice(stuff.end(), newStuff, newStuff.begin(), newStuff.end());
118 }

120 static gboolean invalidate_view(ViewInterface::WeakPtr vi)
121 {
122  ViewInterface::Ptr v = vi.lock();
123  if(v)
124  {
125  gdk_threads_enter();
126  v->invalidate();
127  gdk_threads_leave();
128  }
129  return false;
130 }

138 void TiledBitmapViewData::clearVolatileStuff()
139 {
140  // printf("Erasing volatile stuff for %p\n", viewInterface);
141  std::list<boost::shared_ptr<void>> oldVolatileStuff;
142  {
143  boost::unique_lock<boost::mutex> lock(mut);
144  oldVolatileStuff.swap(volatileStuff);
145  }
146  // oldVolatileStuff gets erased here, without holding the lock. So
147  // if anyone is calling storeVolatileStuff() as a side effect, they
148  // can do so safely.
149  oldVolatileStuff.clear();
150  resetNeededTiles();
151 }

153 void TiledBitmapViewData::tileLoaded(ConstTile::Ptr tile)
154 {
155  boost::unique_lock<boost::mutex> lock(mut);
156  stuff.push_back(tile);
157 
158  // printf("TiledBitmapViewData::tileLoaded: redrawPending=%d\n", redrawPending);
159 
160  if(!redrawPending)
161  {
162  // We're not sure about whether gdk_threads_enter() has been
163  // called or not, so we have no choice but to invalidate on
164  // another thread.
165  Scroom::GtkHelpers::Wrapper w = Scroom::GtkHelpers::wrap(boost::bind(invalidate_view, viewInterface));
166  gtk_idle_add(w.f, w.data);
167  redrawPending = true;
168  }
169 }
