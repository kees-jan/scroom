79  RawPageData::Ptr Blob::load()
80  {
81  RawPageData::Ptr result = weakData.lock();
82  if(!result)
83  {
84  switch(state)
85  {
86  case UNINITIALIZED:
87  // Allocate new data
88  data = static_cast<uint8_t*>(malloc(size * sizeof(uint8_t)));
89  break;
90  case CLEAN:
91  // Decompress data
92  data = static_cast<uint8_t*>(malloc(size * sizeof(uint8_t)));
93  Detail::decompressBlob(data, size, pages, provider);
94  break;
95  case DIRTY:
96  break;
97  case COMPRESSING:
98  // Data is currently being compressed. Abort...
99  state = DIRTY;
100  break;
101  default:
102  printf("PANIC: Illegal state %d\n", state);
103  break;
104  }
105 
106  // data should point to something valid here...
107  result = RawPageData::Ptr(data, UnloadData(shared_from_this<Blob>()));
108  weakData = result;
109  refcount++;
110  }
111 
112  return result;
113  }
