63  LayerSpecResult get(const BitmapMetaData& bmd)
64  {
65  LayerSpecResult ls;
66 
67  // Later registered functions override earlier ones
68  for(const auto& f: boost::adaptors::reverse(functions))
69  {
70  if(!std::get<0>(ls).empty())
71  {
72  break;
73  }
74 
75  ls = f(bmd);
76  }
77 
78  return ls;
79  }
80 
81  private:
82  LayerSpecForBitmapper()
83  : functions({RGBBitmap, CMYKBitmap, GreyscaleBitmap, ColormappedBitmap})
84  {}

87  };

91  LayerSpecResult RGBBitmap(const BitmapMetaData& bmd)

150  LayerSpecResult GreyscaleBitmap(const BitmapMetaData& bmd)
151  {
152  if(bmd.type != Greyscale)
153  {
154  return {};
155  }
156 
157  require(bmd.samplesPerPixel == 1);
158 
159  if(bmd.bitsPerSample != 1 && bmd.bitsPerSample != 2 && bmd.bitsPerSample != 4 && bmd.bitsPerSample != 8)
160  {
161  printf("ERROR: Greyscale bitmaps with %u bits per pixel are not supported (yet)\n", bmd.bitsPerSample);
162  return {};
163  }
164 
165  LayerSpec ls;
166  ColormapHelperBase::Ptr colormapHelper = bmd.colormapHelper;
167  if(!colormapHelper)
168  {
169  colormapHelper = MonochromeColormapHelper::create(2);
170  }
171 
172  if(bmd.bitsPerSample == 2 || bmd.bitsPerSample == 4)
173  {
174  ls.push_back(Operations::create(colormapHelper, bmd.bitsPerSample));
175  ls.push_back(OperationsColormapped::create(colormapHelper, bmd.bitsPerSample));
176  }
177  else
178  {
179  if(bmd.bitsPerSample == 1)
180  {
181  ls.push_back(Operations1bpp::create(colormapHelper));
182  }
183  ls.push_back(Operations8bpp::create(colormapHelper));
184  }
185 
186  return LayerSpecResult(ls, colormapHelper);
187  }
