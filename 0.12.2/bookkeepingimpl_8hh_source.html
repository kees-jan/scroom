203  template <typename K, typename V>
204  inline Token MapBase<K, V>::reserve(const K& k)
205  {
206  boost::mutex::scoped_lock lock(mut);
207  if(map.end() != map.find(k))
208  {
209  throw std::invalid_argument("Key already exists");
210  }
211 
212  typename Detail::ValueType<V>::Ptr pv = Detail::ValueType<V>::create(V());
213  map[k] = pv;
214 
215  Token t = Detail::MapTokenImpl<K, V>::create(shared_from_this<MapBase<K, V>>(), k);
216  t.add(pv);
217  pv->token = t;
218  return t;
219  }

222  inline Token MapBase<K, V>::reReserve(const K& k)
223  {
224  boost::mutex::scoped_lock lock(mut);
225  auto i = map.find(k);
226 
227  if(map.end() == i)
228  {
229  map[k] = typename Detail::ValueType<V>::WeakPtr();
230  i = map.find(k);
231  }
232 
233  typename Detail::ValueType<V>::Ptr pv = i->second.lock();
234  if(!pv)
235  {
236  pv = Detail::ValueType<V>::create(V());
237  i->second = pv;

239 
240  Token t = pv->token.lock();
241  if(!t)

243  t = Detail::MapTokenImpl<K, V>::create(shared_from_this<MapBase<K, V>>(), k);
244  t.add(pv);
245  pv->token = t;
246  }
247 
248  return t;
