115 void MeasureHandler::render(ViewInterface::Ptr const& vi,
116  cairo_t* cr,
117  Scroom::Utils::Rectangle<double> presentationArea,
118  int zoom)
119 {
120  UNUSED(vi);
121 
122  if(selection)
123  {
124  auto aspectRatio = vi->getCurrentPresentation()->getAspectRatio();
125  auto start = Scroom::Utils::Point<int>(selection->start) - presentationArea.getTopLeft();
126  auto end = Scroom::Utils::Point<int>(selection->end) - presentationArea.getTopLeft();
127 
128  if(zoom >= 0)
129  {
130  const int pixelSize = 1 << zoom;
131  start *= pixelSize;
132  start *= aspectRatio;
133  end *= pixelSize;
134  end *= aspectRatio;
135  }
136  else
137  {
138  const int pixelSize = 1 << -zoom;
139  start *= aspectRatio;
140  start /= pixelSize;
141  end *= aspectRatio;
142  end /= pixelSize;
143  }
144 
145  cairo_set_line_width(cr, 1);
146  cairo_set_source_rgb(cr, 0.75, 0, 0); // Dark Red
147  drawCross(cr, start);
148  drawCross(cr, end);
149  cairo_stroke(cr);
150  cairo_set_source_rgb(cr, 1, 0, 0); // Red
151  cairo_move_to(cr, start.x, start.y);
152  cairo_line_to(cr, end.x, end.y);
153  cairo_stroke(cr);
154  }
155 }
