50  static Colormap::Ptr createDefault(int n)
51  {
52  Colormap::Ptr result = create();
53  result->name = "Default";
54  result->colors.reserve(n);
55  result->colors.clear();
56  double max = n - 1;
57  for(int i = 0; i < n; i++)
