44 Scroom::Utils::Stuff OperationsCMYK32::cache(const ConstTile::Ptr tile)
45 {
46  // Allocate the space for the cache - stride is the height of one row
47  const int stride = cairo_format_stride_for_width(CAIRO_FORMAT_ARGB32, tile->width);
48  boost::shared_ptr<uint8_t> data = shared_malloc(static_cast<size_t>(stride * tile->height));
49 
50  // Row is a pointer to a row of pixels (destination)
51  auto* row = reinterpret_cast<uint32_t*>(data.get());
52  // Cur is a pointer to the start of the row in the tile (source)
53  const uint8_t* cur = tile->data.get();
54 
55  for(int i = 0; i < 4 * tile->height * tile->width; i += 4)
56  {
57  // Convert CMYK to ARGB, because cairo doesn't know how to render CMYK.
58  auto C_i = static_cast<uint8_t>(255 - cur[i]);
59  auto M_i = static_cast<uint8_t>(255 - cur[i + 1]);
60  auto Y_i = static_cast<uint8_t>(255 - cur[i + 2]);
61  auto K_i = static_cast<uint8_t>(255 - cur[i + 3]);
62 
63  uint32_t R = static_cast<uint8_t>((C_i * K_i) / 255);
64  uint32_t G = static_cast<uint8_t>((M_i * K_i) / 255);
65  uint32_t B = static_cast<uint8_t>((Y_i * K_i) / 255);
66 
67  // Write 255 as alpha (fully opaque)
68  row[i / 4] = 255u << 24 | R << 16 | G << 8 | B;
69  }
70 
71  return Scroom::Bitmap::BitmapSurface::create(tile->width, tile->height, CAIRO_FORMAT_ARGB32, stride, data);
72 }

74 void OperationsCMYK32::reduce(Tile::Ptr target, const ConstTile::Ptr source, int top_left_x, int top_left_y)
75 {
76  // Reducing by a factor 8
77  int sourceStride = 8 * source->width / 2; // stride in bytes
78  const byte* sourceBase = source->data.get();
79 
80  int targetStride = 8 * target->width / 2; // stride in bytes
81  byte* targetBase = target->data.get() + (target->height * top_left_y + top_left_x) * targetStride / 8;
82 
83  for(int y = 0; y < source->height / 8; y++)
84  {
85  byte* targetPtr = targetBase;
86 
87  for(int x = 0; x < source->width / 8; x++)
88  {
89  // We want to store the average colour of the 8*8 pixel image
90  // with (x, y) as its top-left corner into targetPtr.
91  const byte* base = sourceBase + 8 * 4 * x; // start of the row
92  const byte* end = base + 8 * sourceStride; // end of the row
93 
94  int sum_c = 0;
95  int sum_m = 0;
96  int sum_y = 0;
97  int sum_k = 0;
98  for(const byte* row = base; row < end; row += sourceStride)
99  {
100  for(size_t current = 0; current < 8 * 4; current += 4)
101  {
102  sum_c += row[current];
103  sum_m += row[current + 1];
104  sum_y += row[current + 2];
105  sum_k += row[current + 3];
106  }
107  }
108 
109  targetPtr[0] = static_cast<byte>(sum_c / 64);
110  targetPtr[1] = static_cast<byte>(sum_m / 64);
111  targetPtr[2] = static_cast<byte>(sum_y / 64);
112  targetPtr[3] = static_cast<byte>(sum_k / 64);
113 
114  targetPtr += 4;
115  }
116 
117  targetBase += targetStride;
118  sourceBase += sourceStride * 8;
119  }

336  uint8_t C_i;
337 
338  uint8_t M_i;
339 
340  uint8_t Y_i;
341 
342  uint8_t K_i;
343  if((i & 1) == 0)
344  { // even pixels -> top half of byte
345  C_i = static_cast<uint8_t>(((cur[i / 2]) >> 7) - 1); // 0 -> 255 (= -1), 1 -> 0
346  M_i = static_cast<uint8_t>(((cur[i / 2] & 0x40) >> 6) - 1);
347  Y_i = static_cast<uint8_t>(((cur[i / 2] & 0x20) >> 5) - 1);
348  K_i = static_cast<uint8_t>(((cur[i / 2] & 0x10) >> 4) - 1);
349  }
350  else
351  { // odd pixels -> lower half of the byte
352  C_i = static_cast<uint8_t>(((cur[i / 2] & 0x08) >> 3) - 1); // 0 -> 255 (= -1), 1 -> 0
353  M_i = static_cast<uint8_t>(((cur[i / 2] & 0x04) >> 2) - 1);
354  Y_i = static_cast<uint8_t>(((cur[i / 2] & 0x02) >> 1) - 1);
355  K_i = static_cast<uint8_t>(((cur[i / 2] & 0x01)) - 1);
356  }
357 
358  uint32_t R = static_cast<uint8_t>((C_i * K_i) / 255);
359  uint32_t G = static_cast<uint8_t>((M_i * K_i) / 255);
360  uint32_t B = static_cast<uint8_t>((Y_i * K_i) / 255);
361 
362  // Write 255 as alpha (fully opaque)
