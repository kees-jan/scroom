283 void ThreadPool::work(ThreadPool::PrivateData::Ptr priv)
284 {
285  boost::mutex::scoped_lock lock(priv->mut);
286  while(priv->alive)
287  {
288  if(priv->jobcount > 0)
289  {
290  priv->jobcount--;
291  lock.unlock();
292  do_one(priv);
293  lock.lock();
294  }
295  else
296  {
297  priv->cond.wait(lock);
298  }
299  }
300 
301  bool busy = priv->completeAllJobsBeforeDestruction;
302  while(busy)
303  {
304  if(priv->jobcount > 0)
305  {
306  priv->jobcount--;
307  lock.unlock();
308  do_one(priv);
309  lock.lock();
310  }
311  else
312  {
313  busy = false;
314  }
315  }

318 void ThreadPool::do_one(ThreadPool::PrivateData::Ptr priv)
319 {
320  ThreadPool::Job job;
321 
322  {
323  boost::mutex::scoped_lock lock(priv->mut);
324 
325  while(!priv->jobs.empty() && priv->jobs.begin()->second.empty())
326  {
327  priv->jobs.erase(priv->jobs.begin());
328  }
329 
330  if(!priv->jobs.empty() && !priv->jobs.begin()->second.empty())
331  {
332  job = priv->jobs.begin()->second.front();
333  priv->jobs.begin()->second.pop();
334  }
335  else
336  {
337  printf("PANIC: JobQueue empty while it shouldn't be\n");
338  }
339  }
340 
341  if(job.queue)
342  {
343  QueueLock l(job.queue);
344  if(l.queueExists())
345  {
346  boost::this_thread::disable_interruption while_executing_jobs;
347  job.fn();
348  }
349  }
