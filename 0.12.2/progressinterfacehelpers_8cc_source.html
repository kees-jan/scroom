272  void ProgressInterfaceMultiplexer::updateProgressState()
273  {
274  ProgressStateInterface::State state = ProgressStateInterface::IDLE;
275  double progress = 0.0;
276  int workers = 0;
277 
278  boost::mutex::scoped_lock l(mut);
279  for(const ChildData::Ptr& child: children)
280  {
281  boost::mutex::scoped_lock child_lock(child->mut);
282  switch(child->state)
283  {
284  case ProgressStateInterface::IDLE:
285  break;
286  case ProgressStateInterface::WAITING:
287  if(state == ProgressStateInterface::IDLE || state == ProgressStateInterface::FINISHED)
288  {
289  state = ProgressStateInterface::WAITING;
290  }
291  progress += child->progress;
292  workers++;
293  break;
294  case ProgressStateInterface::WORKING:
295  if(state != ProgressStateInterface::WORKING)
296  {
297  state = ProgressStateInterface::WORKING;
298  }
299  progress += child->progress;
300  workers++;
301  break;
302  case ProgressStateInterface::FINISHED:
303  if(state == ProgressStateInterface::IDLE)
304  {
305  state = ProgressStateInterface::FINISHED;
306  }
307  progress += 1.0;
308  workers++;
309  break;
310  }
311  }
312 
313  parent->setProgress(state, progress / workers);
314 
315  if(state == ProgressStateInterface::FINISHED)
316  {
317  for(const ChildData::Ptr& child: children)
318  {
319  child->clearFinished();
320  }
321  }
322  }
