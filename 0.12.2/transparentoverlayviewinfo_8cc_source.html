159 void TransparentOverlayViewInfo::redraw(cairo_t* cr, Scroom::Utils::Rectangle<double> const& pa, int zoom)
160 {
161  using Scroom::Bitmap::BitmapSurface;
162 
163  GdkRectangle presentationArea = pa.toGdkRectangle();
164 
165  GdkRectangle viewArea;
166  viewArea.x = 0;
167  viewArea.y = 0;
168 
169  if(zoom > 0)
170  {
171  const int pixelSize = 1 << zoom;
172  viewArea.width = presentationArea.width * pixelSize;
173  viewArea.height = presentationArea.height * pixelSize;
174  }
175  else
176  {
177  const int pixelSize = 1 << -zoom;
178  viewArea.width = presentationArea.width / pixelSize;
179  viewArea.height = presentationArea.height / pixelSize;
180  }
181 
182  BitmapSurface::Ptr s = BitmapSurface::create(viewArea.width, viewArea.height, CAIRO_FORMAT_ARGB32);
183  cairo_surface_t* surface = s->get();
184  cairo_t* cr_sub = cairo_create(surface);
185 
186  int count = 0;
187 
188  for(size_t i = 0; i < children.size(); i++)
189  {
190  if(gtk_toggle_button_get_active(GTK_TOGGLE_BUTTON(buttons[i])))
191  {
192  PresentationInterface::Ptr& p = children[i];
193 
194  Colormappable::Ptr c = boost::dynamic_pointer_cast<Colormappable>(p);
195  bool hasTransparentBackground = false;
196  if(c && p->isPropertyDefined(TRANSPARENT_BACKGROUND_PROPERTY_NAME))
197  {
198  if(count == 0)
199  {
200  c->disableTransparentBackground();
201  }
202  else
203  {
204  c->setTransparentBackground();
205  hasTransparentBackground = true;
206  }
207  }
208 
209  cairo_save(cr_sub);
210  cairo_set_operator(cr_sub, CAIRO_OPERATOR_CLEAR);
211  cairo_paint(cr_sub);
212  cairo_restore(cr_sub);
213 
214  cairo_save(cr_sub);
215  p->redraw(childViews[p], cr_sub, presentationArea, zoom);
216  cairo_restore(cr_sub);
217 
218  cairo_set_source_surface(cr, surface, 0, 0);
219 
220  if(hasTransparentBackground)
221  {
222  cairo_paint(cr);
223  }
224  else
225  {
226  cairo_paint_with_alpha(cr, 1.0 / (count + 1));
227  }
228 
229  if(!hasTransparentBackground)
230  {
231  count++;
232  }
233  }
234  }
235 
236  cairo_destroy(cr_sub);
237 }
