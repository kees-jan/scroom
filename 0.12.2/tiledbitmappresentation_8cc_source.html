211  PipetteLayerOperations::PipetteColor TiledBitmapPresentation::getPixelAverages(Scroom::Utils::Rectangle<int> area)
212  {
213  require(pipetteLayerOperation);
214 
215  Scroom::Utils::Rectangle<int> presentationArea = getRect().toIntRectangle();
216  area = area.intersection(presentationArea);
217 
218  Layer::Ptr bottomLayer = tbi->getBottomLayer();
219  PipetteLayerOperations::PipetteColor pipetteColors;
220 
221  int totalPixels = area.getWidth() * area.getHeight();
222 
223  if(totalPixels == 0)
224  {
225  return {};
226  }
227 
228  // Get start tile (tile_pos_x_start, tile_pos_y_start)
229  int tile_pos_x_start = area.getLeft() / TILESIZE;
230  int tile_pos_y_start = area.getTop() / TILESIZE;
231 
232  // Get end tile (tile_pos_x_end, tile_pos_y_end)
233  int tile_pos_x_end = (area.getRight() - 1) / TILESIZE;
234  int tile_pos_y_end = (area.getBottom() - 1) / TILESIZE;
235 
236  for(int x = tile_pos_x_start; x <= tile_pos_x_end; x++)
237  {
238  for(int y = tile_pos_y_start; y <= tile_pos_y_end; y++)
239  {
240  ConstTile::Ptr tile = bottomLayer->getTile(x, y)->getConstTileSync();
241  Scroom::Utils::Rectangle<int> tile_rectangle(x * TILESIZE, y * TILESIZE, tile->width, tile->height);
242 
243  Scroom::Utils::Rectangle<int> inter_rect = tile_rectangle.intersection(area);
244  Scroom::Utils::Point<int> base(x * TILESIZE, y * TILESIZE);
245 
246  inter_rect -= base; // rectangle coordinates relative to constTile with topleft corner (0,0)
247 
248  pipetteColors = sumPipetteColors(pipetteColors, pipetteLayerOperation->sumPixelValues(inter_rect, tile));
249  }
250  }
251  return dividePipetteColors(pipetteColors, totalPixels);
252  }

348  PresentationInterface::Ptr OpenTiledBitmapAsPresentation::open(const std::string& fileName)
349  {
350  auto t = openTiledBitmapInterface->open(fileName);
351  BitmapMetaData bmd = std::move(std::get<0>(t));
352  Layer::Ptr bottomLayer = std::move(std::get<1>(t));
353  ReloadFunction load = std::move(std::get<2>(t));
354 
355  auto lsr = LayerSpecForBitmap(bmd);
356  LayerSpec layerSpec = std::move(std::get<0>(lsr));
357  ColormapHelperBase::Ptr colormapHelper = std::move(std::get<1>(lsr));
358 
359  PresentationInterface::Ptr result;
360  if(bottomLayer && !layerSpec.empty())
361  {
362  auto tiledBitmap = TiledBitmap::create(bottomLayer, layerSpec);
363  PipetteLayerOperations::Ptr pipetteLayerOperation = boost::dynamic_pointer_cast<PipetteLayerOperations>(layerSpec[0]);
364 
365  std::map<std::string, std::string> properties;
366  if(bmd.colormapHelper)
367  {
368  properties = bmd.colormapHelper->getProperties();
369  }
370  if(pipetteLayerOperation)
371  {
372  properties[PIPETTE_PROPERTY_NAME] = "";
373  }
374 
375  auto tiledBitmapPresentation =
376  TiledBitmapPresentation::create(fileName, bmd.rect, tiledBitmap, properties, colormapHelper, pipetteLayerOperation);
377  tiledBitmapPresentation->add(load(tiledBitmap->progressInterface()));
378 
379  if(bmd.aspectRatio)
380  {
381  result = TransformPresentation::create(tiledBitmapPresentation, TransformationData::create(*bmd.aspectRatio));
382  }
383  else
384  {
385  result = tiledBitmapPresentation;
386  }
387  }
388  return result;
389  }
