131 ConstTile::Ptr CompressedTile::do_load()
132 {
133  bool didLoad = false;
134 
135  ConstTile::Ptr result;
136  {
137  boost::mutex::scoped_lock lock(stateData);
138  cleanupState();
139  state = TSI_LOADING_SYNCHRONOUSLY;
140  }
141  {
142  boost::mutex::scoped_lock lock(tileData);
143  result = constTile.lock(); // This ought to fail
144  if(!result)
145  {
146  result = ConstTile::Ptr(new ConstTile(TILESIZE, TILESIZE, bpp, data->getConst()));
147  constTile = result;
148  didLoad = true;
149  }
150  }
151  {
152  boost::mutex::scoped_lock lock(stateData);
153  cleanupState();
154  state = TSI_NORMAL;
155  if(didLoad)
156  {
157  notifyObservers(result);
158  }
159  }
160 
161  return result;
162 }

177 TileState CompressedTile::getState()
178 {
179  TileState result = TILE_UNINITIALIZED;
180  switch(state)
181  {
182  case TSI_NORMAL:
183  {
184  Tile::Ptr t = tile.lock();
185  if(t)
186  {
187  result = TILE_LOADED;
188  }
189  else
190  {
191  result = TILE_UNLOADED;
192  }
193  }
194  break;
195  case TSI_OUT_OF_BOUNDS:
196  result = TILE_OUT_OF_BOUNDS;
197  break;
198  case TSI_LOADING_SYNCHRONOUSLY:
199  case TSI_LOADING_ASYNCHRONOUSLY:
200  result = TILE_UNLOADED;
201  break;
202  case TSI_UNINITIALIZED:
203  result = TILE_UNINITIALIZED;
204  break;
205  }
206  return result;
207 }

214 void CompressedTile::observerAdded(TileLoadingObserver::Ptr const& observer, Scroom::Bookkeeping::Token const& token)
215 {
216  ConstTile::Ptr result = constTile.lock();
217  ThreadPool::Queue::Ptr queue_ = queue.lock();
218 
219  if(!result)
220  {
221  boost::mutex::scoped_lock lock(stateData);
222  cleanupState();
223  switch(state)
224  {
225  case TSI_UNINITIALIZED:
226  case TSI_LOADING_ASYNCHRONOUSLY:
227  case TSI_LOADING_SYNCHRONOUSLY:
228  // Need to do nothing. All will be well.
229  break;
230 
231  default:
232  case TSI_OUT_OF_BOUNDS:
233  // Shouldn't happen
234  break;
235 
236  case TSI_NORMAL:
237  // Start an asynchronous load
238  if(!queue_)
239  {
240  queue_ = ThreadPool::Queue::create();
241  queue = queue_;
242  }
243  CpuBound()->schedule(boost::bind(&CompressedTile::do_load, this), LOAD_PRIO, queue_);
244  state = TSI_LOADING_ASYNCHRONOUSLY;
245 
246  break;
247  }
248  }
249 
250  // When the last observer goes away, cancel the load
251  if(queue_)
252  {
253  token.add(queue_);
254  }
255 
256  if(result)
257  {
258  observer->tileLoaded(result);
259  }
260 }
