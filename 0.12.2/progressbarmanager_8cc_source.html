78  bool ProgressBarPulser::doWork()
79  {
80  // Locking these the other way around results in a deadlock. See ticket #40
81  Scroom::GtkHelpers::TakeGdkLock gdkLock;
82  boost::mutex::scoped_lock lock(mut);
83 
84  while(current == progressbars.end() || *current == NULL)
85  {
86  if(progressbars.empty())
87  {
88  return false;
89  }
90  else if(current == progressbars.end())
91  {
92  current = progressbars.begin();
93  }
94  else if(*current == NULL)
95  {

97  }
98  else
99  {
100  defect();
101  }
102  }
103 
104  gtk_progress_bar_pulse(*current);
