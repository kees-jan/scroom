23 static Scroom::MemoryBlobs::PageProvider::Ptr createProvider(double width, double height, int bpp)
24 {
25  double tileCount = (width * height) / (TILESIZE * TILESIZE);
26  double tileSize = (bpp / 8.0) * TILESIZE * TILESIZE;
27 
28  double guessedTileSizeAfterCompression = tileSize / 100;
29  const int pagesize = 4096;
30  int pagesPerBlock = std::max(int(ceil(guessedTileSizeAfterCompression / 10 / pagesize)), 1);
31 
32  int blockSize = pagesPerBlock * pagesize;
33  int blockCount = std::max(int(ceil(tileCount / 10)), 64);
34 
35  printf("Creating a PageProvider providing %d blocks of %d bytes\n", blockCount, blockSize);
36  return Scroom::MemoryBlobs::PageProvider::create(blockCount, blockSize);
37 }

207 void DataFetcher::operator()()
208 {
209  // printf("Attempting to fetch bitmap data for tileRow %d...\n", currentRow);
210  QueueJumper::Ptr qj = QueueJumper::create();
211 
212  threadPool->schedule(qj, REDUCE_PRIO, queue);
213 
214  CompressedTileLine& tileLine = layer->getTileLine(currentRow);
215  std::vector<Tile::Ptr> tiles;
216  for(int x = 0; x < horTileCount; x++)
217  {
218  CompressedTile::Ptr ti = tileLine[x];
219  Scroom::Utils::Stuff s = ti->initialize();
220  tiles.push_back(ti->getTileSync());
221  }
222  int lineCount = std::min(TILESIZE, height - currentRow * TILESIZE);
223 
224  sp->fillTiles(currentRow * TILESIZE, lineCount, TILESIZE, 0, tiles);
225 
226  for(int x = 0; x < horTileCount; x++)
227  {
228  tileLine[x]->reportFinished();
229  }
230 
231  currentRow++;
232  if(currentRow < verTileCount)
233  {
234  DataFetcher successor(*this);
235  if(!qj->setWork(successor))
236  {
237  threadPool->schedule(successor, DATAFETCH_PRIO, queue);
238  }
239  }
240  else
241  {
242  sp->done();
243  on_finished();
244  }
245 }
