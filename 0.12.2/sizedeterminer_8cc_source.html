17  Scroom::Utils::Rectangle<double> DetermineSize(std::list<PresentationInterface::Ptr> presentations)
18  {
19  double left = std::numeric_limits<double>::max();
20  double top = std::numeric_limits<double>::max();
21  double right = std::numeric_limits<double>::min();
22  double bottom = std::numeric_limits<double>::min();
23 
24  for(PresentationInterface::Ptr const& p: presentations)
25  {
26  Scroom::Utils::Rectangle<double> rect = p->getRect();
27  left = std::min(left, rect.getLeft());
28  top = std::min(top, rect.getTop());
29  right = std::max(right, rect.getRight());
30  bottom = std::max(bottom, rect.getBottom());
31  }
32  return Scroom::Utils::Rectangle<double>(left, top, right - left, bottom - top);
33  }
