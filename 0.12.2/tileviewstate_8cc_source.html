135 void TileViewState::process(ThreadPool::WeakQueue::Ptr wq)
136 {
137  for(;;)
138  {
139  boost::mutex::scoped_lock l(mut);
140  // printf("process: Tile %d, %d, %d: state=%d, desired=%d, queue? %d\n",
141  // parent->depth, parent->x, parent->y, state, desiredState, (bool)(wq==weakQueue));
142 
143  if(wq == weakQueue && state < desiredState)
144  {
145  boost::function<void()> fn;
146 
147  switch(state)
148  {
149  case LOADED:
150  fn = boost::bind(&TileViewState::computeBase, shared_from_this<TileViewState>(), wq, tile, lo);
151  state = COMPUTING_BASE;
152  break;
153 
154  case BASE_COMPUTED:
155  fn = boost::bind(&TileViewState::computeZoom, shared_from_this<TileViewState>(), wq, tile, lo, baseCache, zoom);
156  state = COMPUTING_ZOOM;
157  break;
158 
159  case ZOOM_COMPUTED:
160  state = DONE;
161  fn = boost::bind(&TileViewState::reportDone, shared_from_this<TileViewState>(), wq, tile);
162  break;
163 
164  case INIT:
165  case COMPUTING_BASE:
166  case COMPUTING_ZOOM:
167  case DONE:
168  default:
169  printf("PANIC: Don't know what to do in state %d\n", state);
170  return;
171  }
172 
173  l.unlock();
174  fn();
175  }
176  else
177  {
178  break;
179  }
180  }
181 
182  {
183  boost::mutex::scoped_lock l(mut);
184 
185  if(wq == weakQueue)
186  {
187  queue.reset();
188  weakQueue.reset();
189  }
190  }
191 }
