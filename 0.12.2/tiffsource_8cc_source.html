89  boost::optional<Scroom::Utils::Point<double>> getAspectRatio(const TIFFPtr& tif)
90  {
91  float resolutionX{};
92  float resolutionY{};
93  uint16 resolutionUnit{};
94 
95  if(TIFFGetField(tif.get(), TIFFTAG_XRESOLUTION, &resolutionX) && TIFFGetField(tif.get(), TIFFTAG_YRESOLUTION, &resolutionY)
96  && TIFFGetField(tif.get(), TIFFTAG_RESOLUTIONUNIT, &resolutionUnit))
97  {
98  if(resolutionUnit != RESUNIT_NONE)
99  {
100  // Fix aspect ratio only
101  float base = std::max(resolutionX, resolutionY);
102  resolutionX = base / resolutionX;
103  resolutionY = base / resolutionY;
104  }
105  return Scroom::Utils::make_point<double>(resolutionX, resolutionY);
106  }
107 
108  return {};

117  boost::optional<std::tuple<Scroom::TiledBitmap::BitmapMetaData, TIFFPtr>> open(const std::string& fileName)
118  {
119  try
120  {
121  TIFFPtr tif(TIFFOpen(fileName.c_str(), "r"), &TIFFCloseUnlessNull);
122  if(!tif)
123  {
124  // Todo: report error
125  printf("PANIC: Failed to open file %s\n", fileName.c_str());
126  return {};
127  }
128 
129  auto spp = TIFFGetFieldCheckedOr<uint16_t>(tif, TT(TIFFTAG_SAMPLESPERPIXEL), 1); // Default value, according to tiff spec
130  auto bps = TIFFGetFieldCheckedOr<uint16_t>(tif, TT(TIFFTAG_BITSPERSAMPLE), (spp == 1) ? 1 : 8);
131  auto width = TIFFGetFieldChecked<uint32_t>(tif, TT(TIFFTAG_IMAGEWIDTH));
132  auto height = TIFFGetFieldChecked<uint32_t>(tif, TT(TIFFTAG_IMAGELENGTH));
133  auto photometric = TIFFGetFieldChecked<uint16_t>(tif, TT(TIFFTAG_PHOTOMETRIC));
134 
135  ColormapHelperBase::Ptr colormapHelper = getColormapHelper(tif, bps);
136 
137  if(photometric != PHOTOMETRIC_PALETTE && colormapHelper)
138  {
139  printf("WARNING: Tiff contains a colormap, but photometric isn't palette\n");
140  colormapHelper.reset();
141  }
142 
143  auto numberOfGreyscaleColors = (bps == 2 || bps == 4) ? (1 << bps) : 2; // Yuk
144  switch(photometric)
145  {
146  case PHOTOMETRIC_MINISBLACK:
147  colormapHelper = MonochromeColormapHelper::create(numberOfGreyscaleColors);
148  break;
149 
150  case PHOTOMETRIC_MINISWHITE:
151  colormapHelper = MonochromeColormapHelper::createInverted(numberOfGreyscaleColors);
152  break;
153 
154  case PHOTOMETRIC_PALETTE:
155  if(!colormapHelper)
156  {
157  printf("WEIRD: Photometric is palette, but tiff doesn't contain a colormap\n");
158  colormapHelper = ColormapHelper::create(1 << bps);
159  }
160  break;
161 
162  case PHOTOMETRIC_RGB:
163  case PHOTOMETRIC_SEPARATED:
164  break;
165 
166  default:
167  printf("ERROR: Unrecognized value %d for photometric\n", photometric);
168  return {};
169  }
170 
171  auto aspectRatio = getAspectRatio(tif);
172  if(aspectRatio)
173  {
174  printf("This bitmap has size %u*%u, aspect ratio %.1f*%.1f\n", width, height, aspectRatio->x, aspectRatio->y);
175  }
176  else
177  {
178  printf("This bitmap has size %u*%u\n", width, height);
179  }
180 
181  BitmapMetaData bmd{{}, bps, spp, Scroom::Utils::make_rect<int>(0, 0, width, height), aspectRatio, colormapHelper};
182 
183  if(bps != 1 && bps != 2 && bps != 4 && bps != 8)
184  {
185  printf("ERROR: %u bits per sample not supported (yet)\n", bps);
186  return {};
187  }
188 
189  if(spp == 4)
190  {
191  bmd.type = CMYK;
192  }
193  else if(spp == 3)
194  {
195  bmd.type = RGB;
196 
197  if(bps != 8)
198  {
199  printf("ERROR: A RGB bitmap with %u samples per pixel isn't supported (yet)", bps);
200  return {};
201  }
202  }
203  else if(spp == 1)
204  {
205  bmd.type = (photometric == PHOTOMETRIC_PALETTE) ? Colormapped : Greyscale;
206  }
207  else
208  {
209  printf("PANIC: %u samples per pixel not supported (yet)\n", spp);
210  return {};
211  }
212 
213  return std::make_tuple(bmd, tif);
214  }
215  catch(const std::exception& ex)
216  {
217  printf("PANIC: %s\n", ex.what());
218  return {};
219  }
220  }

233  bool Source::reset()
234  {
235  tif = preOpenedTif;
236  preOpenedTif = nullptr;
237 
238  if(!tif)
239  {
240  auto r = Scroom::Tiff::open(fileName);
241  if(r && approx(std::get<0>(*r), bmd))
242  {
243  tif = std::get<1>(*r);
244  }
245  else
246  {
247  if(r)
248  {
249  printf("ERROR: Can't reload. Bitmap changed too much\n");
250  printf("INFO: Previously: %s\n", to_string(bmd).c_str());
251  printf("INFO: Now: %s\n", to_string(std::get<0>(*r)).c_str());
252  }
253  // if (!r) then the error has already been reported by open()
254  return false;
255  }
256  }
257  ensure(tif);
258 
259  return true;

262  void Source::fillTiles(int startLine, int lineCount, int tileWidth, int firstTile, std::vector<Tile::Ptr>& tiles)
263  {
264  // printf("Filling lines %d to %d, tile %d to %d (tileWidth = %d)\n",
265  // startLine, startLine+lineCount,
266  // firstTile, (int)(firstTile+tiles.size()),
267  // tileWidth);
268  auto spp = bmd.samplesPerPixel;
269  auto bps = bmd.bitsPerSample;
270 
271  const auto startLine_ = static_cast<uint32>(startLine);
272  const auto firstTile_ = static_cast<size_t>(firstTile);
273  const auto scanLineSize = static_cast<size_t>(TIFFScanlineSize(tif.get()));
274  const auto tileStride = static_cast<size_t>(tileWidth * spp * bps / 8);
275  std::vector<byte> row(scanLineSize);
276 
277  const size_t tileCount = tiles.size();
278  auto dataPtr = std::vector<byte*>(tileCount);
279  for(size_t tile = 0; tile < tileCount; tile++)
280  {
281  dataPtr[tile] = tiles[tile]->data.get();
282  }
283 
284  for(size_t i = 0; i < static_cast<size_t>(lineCount); i++)
285  {
286  TIFFReadScanline(tif.get(), row.data(), static_cast<uint32>(i) + startLine_);
287 
288  for(size_t tile = 0; tile < tileCount - 1; tile++)
289  {
290  memcpy(dataPtr[tile], row.data() + (firstTile_ + tile) * tileStride, tileStride);
291  dataPtr[tile] += tileStride;
292  }
293  memcpy(dataPtr[tileCount - 1],
294  row.data() + (firstTile_ + tileCount - 1) * tileStride,
295  scanLineSize - (firstTile_ + tileCount - 1) * tileStride);
296  dataPtr[tileCount - 1] += tileStride;
297  }
298  }
