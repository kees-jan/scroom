30 Scroom::Utils::Stuff Scroom::TiledBitmap::scheduleLoadingBitmap(const SourcePresentation::Ptr& sp,
31  const Layer::Ptr& layer,
32  const ProgressInterface::Ptr& progress)
33 {
34  auto wait_until_done = boost::make_shared<Scroom::Semaphore>();
35  auto queue = ThreadPool::Queue::createAsync();
36  auto weakQueue = queue->getWeak();
37  auto abort = [wait_until_done, queue]() mutable {
38  queue.reset();
39  wait_until_done->V();
40  };
41 
42  auto on_finished = [wait_until_done] { wait_until_done->V(); };
43  progress->setWaiting();
44 
45  Sequentially()->schedule([progress, layer, sp, weakQueue, on_finished, wait_until_done] {
46  gdk_threads_enter();
47  progress->setWorking(0);
48  gdk_threads_leave();
49  layer->fetchData(sp, weakQueue, on_finished);
50  wait_until_done->P();
51  });
52 
53  return Scroom::Utils::on_destruction(abort);
54 }

89 void TiledBitmap::initialize(const Layer::Ptr& bottom)
90 {
91  int width = bitmapWidth;
92  int height = bitmapHeight;
93  unsigned int i = 0;
94  LayerOperations::Ptr lo = ls[i];
95  Scroom::MemoryBlobs::PageProvider::Ptr provider = bottom->getPageProvider();
96 
97  registrations.push_back(bottom->registerObserver(shared_from_this<TileInitialisationObserver>()));
98  layers.push_back(bottom);
99 
100  Layer::Ptr prevLayer = bottom;
101  LayerOperations::Ptr prevLo = lo;
102 
103  width = (width + 7) / 8; // Round up
104  height = (height + 7) / 8;
105  i++;
106 
107  while(std::max(width, height) > TILESIZE / 4)
108  {
109  if(i < ls.size())
110  {
111  lo = ls[i];
112  }
113 
114  Layer::Ptr layer = Layer::create(i, width, height, lo->getBpp(), provider);
115  registrations.push_back(layer->registerObserver(shared_from_this<TileInitialisationObserver>()));
116  layers.push_back(layer);
117 
118  connect(layer, prevLayer, prevLo);
119 
120  prevLayer = layer;
121  prevLo = lo;
122  width = (width + 7) / 8; // Round up
123  height = (height + 7) / 8;
124  i++;
125  }
126 }

138 void TiledBitmap::connect(Layer::Ptr const& layer, Layer::Ptr const& prevLayer, const LayerOperations::Ptr& prevLo)
139 {
140  int horTileCount = prevLayer->getHorTileCount();
141  int verTileCount = prevLayer->getVerTileCount();
142 
143  std::vector<LayerCoordinator::Ptr> coordinators_;
144 
145  for(int j = 0; j < verTileCount; j++)
146  {
147  int voffset = j % 8;
148  if(!voffset)
149  {
150  // New line of target tiles
151  coordinators_.clear();
152  CompressedTileLine& tileLine = layer->getTileLine(j / 8);
153  for(auto& z: tileLine)
154  {
155  LayerCoordinator::Ptr lc = LayerCoordinator::create(z, prevLo);
156  coordinators_.push_back(lc);
157  coordinators.push_back(lc);
158  }
159  }
160 
161  for(int i = 0; i < horTileCount; i++)
162  {
163  int hoffset = i % 8;
164  LayerCoordinator::Ptr lc = coordinators_[i / 8];
165  lc->addSourceTile(hoffset, voffset, prevLayer->getTile(i, j));
166  }
167  }
168 }

198 void TiledBitmap::redraw(ViewInterface::Ptr const& vi,
199  cairo_t* cr,
200  Scroom::Utils::Rectangle<double> const& presentationArea,
201  int zoom)
202 {
203  TiledBitmapViewData::Ptr viewData_ = viewData[vi];
204  auto scaledRequestedPresentationArea = presentationArea;
205 
206  unsigned int layerNr = 0;
207  while(zoom <= -3 && layerNr < layers.size() - 1)
208  {
209  layerNr++;
210  zoom += 3;
211  scaledRequestedPresentationArea /= 8;
212  }
213  Layer::Ptr layer = layers[layerNr];
214  LayerOperations::Ptr layerOperations = ls[std::min(ls.size() - 1, static_cast<size_t>(layerNr))];
215 
216  const Scroom::Utils::Rectangle<int> actualPresentationArea = layer->getRect();
217  const auto validPresentationArea = scaledRequestedPresentationArea.intersection(actualPresentationArea);
218 
219  const int left = scaledRequestedPresentationArea.getLeft();
220  const int top = scaledRequestedPresentationArea.getTop();
221  const int right = scaledRequestedPresentationArea.getRight();
222  const int bottom = scaledRequestedPresentationArea.getBottom();
223 
224  const int imin = std::max(0, left / TILESIZE);
225  const int imax = (right + TILESIZE - 1) / TILESIZE;
226  const int jmin = std::max(0, top / TILESIZE);
227  const int jmax = (bottom + TILESIZE - 1) / TILESIZE;
228 
229  viewData_->setNeededTiles(layer, imin, imax, jmin, jmax, zoom, layerOperations);
230 
231  const double pixelSize = pixelSizeFromZoom(zoom);
232 
233  layerOperations->initializeCairo(cr);
234 
235  const auto clippedRequestedPresentationArea =
236  scaledRequestedPresentationArea.above(validPresentationArea.getBottom()).leftOf(validPresentationArea.getRight());
237 
238  for(int i = imin; i < imax; i++)
239  {
240  for(int j = jmin; j < jmax; j++)
241  {
242  Scroom::Utils::Point<int> tileIndex(i, j);
243 
244  const auto tileArea = TileAreaForIndex(tileIndex);
245  const auto visibleTileArea = tileArea.intersection(clippedRequestedPresentationArea);
246 
247  const auto tileAreaRect = visibleTileArea - tileArea.getTopLeft();
248  const auto viewAreaRect = (visibleTileArea - clippedRequestedPresentationArea.getTopLeft()) * pixelSize;
249 
250  CompressedTile::Ptr tile = layer->getTile(i, j);
251  TileViewState::Ptr tileViewState = tile->getViewState(vi);
252  Scroom::Utils::Stuff cacheResult = tileViewState->getCacheResult();
253  ConstTile::Ptr t = tile->getConstTileAsync();
254 
255  if(t)
256  {
257  cairo_save(cr);
258  layerOperations->draw(cr, t, tileAreaRect, viewAreaRect, zoom, cacheResult);
259  cairo_restore(cr);
260  }
261  else
262  {
263  layerOperations->drawState(cr, tile->getState(), viewAreaRect);
264  }
265 #ifdef DEBUG_TILES
266  drawTile(cr, tile, viewAreaRect);
267 #endif
268  }
269  }
270 }
