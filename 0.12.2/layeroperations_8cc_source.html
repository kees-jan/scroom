53 BitCountLut::BitCountLut()
54 {
55  for(int i = 0; i < 256; i++)
56  {
57  int sum = 0;
58  for(int v = i; v; v >>= 1)
59  {
60  sum += v & 1;
61  }
62  lut[i] = sum;
63  }

132 Scroom::Utils::Stuff CommonOperations::cacheZoom(const ConstTile::Ptr tile, int zoom, Scroom::Utils::Stuff cache)
133 {
134  // In: Cairo surface at zoom level 0
135  // Out: Cairo surface at requested zoom level
136  Scroom::Utils::Stuff result;
137  if(zoom >= 0)
138  {
139  // Don't zoom in. It is a waste of space
140  result = cache;
141  }
142  else if(!cache)
143  {
144  printf("PANIC: Base caching failed to return anything\n");
145  }
146  else
147  {
148  int divider = 1 << -zoom;
149  BitmapSurface::Ptr source = boost::static_pointer_cast<BitmapSurface>(cache);
150  BitmapSurface::Ptr target = BitmapSurface::create(tile->width / divider, tile->height / divider, CAIRO_FORMAT_ARGB32);
151  result = target;
152 
153  cairo_surface_t* surface = target->get();
154  cairo_t* cr = cairo_create(surface);
155  initializeCairo(cr);
156  cairo_scale(cr, 1.0 / divider, 1.0 / divider);
157  cairo_set_source_surface(cr, source->get(), 0, 0);
158  cairo_paint(cr);
159 
160  cairo_destroy(cr);
161  }
162 
163  return result;
164 }

166 void CommonOperations::draw(cairo_t* cr,
167  const ConstTile::Ptr tile,
168  Scroom::Utils::Rectangle<double> tileArea,
169  Scroom::Utils::Rectangle<double> viewArea,
170  int zoom,
171  Scroom::Utils::Stuff cache)
172 {
173  // In: Cairo surface at requested zoom level
174  // Out: given surface rendered to the canvas
175  UNUSED(tile);
176 
177  setClip(cr, viewArea);
178 
179  if(!cache)
180  {
181  drawState(cr, TILE_UNLOADED, viewArea);
182  }
183  else
184  {
185  BitmapSurface::Ptr source = boost::static_pointer_cast<BitmapSurface>(cache);
186 
187  if(zoom > 0)
188  {
189  // Ask Cairo to zoom in for us
190  int multiplier = 1 << zoom;
191  auto origin = viewArea.getTopLeft() / multiplier - tileArea.getTopLeft();
192 
193  cairo_scale(cr, multiplier, multiplier);
194  cairo_set_source_surface(cr, source->get(), origin.x, origin.y);
195  cairo_pattern_set_filter(cairo_get_source(cr), CAIRO_FILTER_NEAREST);
196  cairo_paint(cr);
197  }
198  else
199  {
200  // Cached bitmap is to scale
201  int divider = 1 << -zoom;
202 
203  auto origin = viewArea.getTopLeft() - tileArea.getTopLeft() / divider;
204  cairo_set_source_surface(cr, source->get(), origin.x, origin.y);
205  cairo_paint(cr);
206  }
207  }

212 {

226  {

299 void Operations1bpp::reduce(Tile::Ptr target, const ConstTile::Ptr source, int x, int y)
300 {
301  // Reducing by a factor 8. Source tile is 1bpp. Target tile is 8bpp
302  int sourceStride = source->width / 8;
303  const byte* sourceBase = source->data.get();
304 
305  int targetStride = target->width;
306  byte* targetBase = target->data.get() + target->height * y * targetStride / 8 + target->width * x / 8;
307 
308  for(int j = 0; j < source->height / 8; j++, targetBase += targetStride, sourceBase += sourceStride * 8)
309  {
310  // Iterate vertically over target
311  const byte* sourcePtr = sourceBase;
312  byte* targetPtr = targetBase;
313 
314  for(int i = 0; i < source->width / 8; i++, sourcePtr++, targetPtr++)
315  {
316  // Iterate horizontally over target
317 
318  // Goal is to compute a 8-bit grey value from a 8*8 black/white
319  // image. To do so, we take each of the 8 bytes, count the
320  // number of 1's in each, and add them. Finally, we divide that
321  // by 64 (the maximum number of ones in that area
322 
323  const byte* current = sourcePtr;
324  int sum = 0;
325  for(int k = 0; k < 8; k++, current += sourceStride)
326  {
327  sum += bcl.lookup(*current);
328  }
329 
330  *targetPtr = sum * 255 / 64;
331  }
332  }
333 }

613 void Operations::reduce(Tile::Ptr target, const ConstTile::Ptr source, int x, int y)
614 {
615  // Reducing by a factor 8. Target is 2*bpp and expects two indices into the colormap
616  int sourceStride = source->width / pixelsPerByte;
617  const byte* sourceBase = source->data.get();
618 
619  const int targetMultiplier = 2; // target is 2*bpp
620  int targetStride = targetMultiplier * target->width / pixelsPerByte;
621  byte* targetBase =
622  target->data.get() + target->height * targetStride * y / 8 + targetMultiplier * target->width * x / 8 / pixelsPerByte;
623 
624  for(int j = 0; j < source->height / 8; j++, targetBase += targetStride, sourceBase += sourceStride * 8)
625  {
626  // Iterate vertically over target
627  const byte* sourcePtr = sourceBase;
628  SampleIterator<uint16_t> targetPtr(reinterpret_cast<uint16_t*>(targetBase), 0, targetMultiplier * bpp);
629 
630  for(int i = 0; i < source->width / 8; i++, sourcePtr += 8 / pixelsPerByte, ++targetPtr)
631  {
632  // Iterate horizontally over target
633 
634  // Goal is to determine which values occurs most often in a 8*8
635  // rectangle, and pick the top two.
636  const byte* base = sourcePtr;
637  byte lookup[pixelMask + 1];
638  memset(lookup, 0, sizeof(lookup));
639 
640  for(int k = 0; k < 8; k++, base += sourceStride)
641  {
642  SampleIterator<const byte> current(base, 0, bpp);
643  for(int l = 0; l < 8; l++, ++current)
644  {
645  ++(lookup[*current]);
646  }
647  }
648  unsigned first = 0;
649  unsigned second = 1;
650  if(lookup[1] > lookup[0])
651  {
652  first = 1;
653  second = 0;
654  }
655  for(unsigned c = 2; c < pixelMask + 1; c++)
656  {
657  if(lookup[c] > lookup[first])
658  {
659  second = first;
660  first = c;
661  }
662  else if(lookup[c] > lookup[second])
663  {
664  second = c;
665  }
666  }
667  if(lookup[second] == 0)
668  {
669  second = first;
670  }
671 
672  targetPtr.set(first << pixelOffset | second);
673  }
674  }

759 void OperationsColormapped::reduce(Tile::Ptr target, const ConstTile::Ptr source, int x, int y)
760 {
761  // Reducing by a factor 8. Source and target both 2*bpp, containing 2 colors
762  const int multiplier = 2; // data is 2*bpp, containing 2 colors
763  int sourceStride = multiplier * source->width / pixelsPerByte;
764  const byte* sourceBase = source->data.get();
765 
766  int targetStride = multiplier * target->width / pixelsPerByte;
767  byte* targetBase =
768  target->data.get() + target->height * y * targetStride / 8 + multiplier * target->width * x / 8 / pixelsPerByte;
769 
770  for(int j = 0; j < source->height / 8; j++, targetBase += targetStride, sourceBase += sourceStride * 8)
771  {
772  // Iterate vertically over target
773  const byte* sourcePtr = sourceBase;
774  SampleIterator<uint16_t> targetPtr(reinterpret_cast<uint16_t*>(targetBase), 0, multiplier * bpp);
775 
776  for(int i = 0; i < source->width / 8; i++, sourcePtr += 8 * multiplier / pixelsPerByte, ++targetPtr)
777  {
778  // Iterate horizontally over target
779 
780  // Goal is to determine which value occurs most often in a 8*8
781  // rectangle, and pick that value.
782  const byte* base = sourcePtr;
783  byte lookup[pixelMask + 1];
784  memset(lookup, 0, sizeof(lookup));
785 
786  for(int k = 0; k < 8; k++, base += sourceStride)
787  {
788  SampleIterator<const uint16_t> current(reinterpret_cast<uint16_t const*>(base), 0, multiplier * bpp);
789  for(int l = 0; l < 8; l++, ++current)
790  {
791  ++lookup[*current & pixelMask];
792  ++lookup[*current >> pixelOffset];
793  }
794  }
795  unsigned first = 0;
796  unsigned second = 1;
797  if(lookup[1] > lookup[0])
798  {
799  first = 1;
800  second = 0;
801  }
802  for(unsigned c = 2; c < pixelMask + 1; c++)
803  {
804  if(lookup[c] > lookup[first])
805  {
806  second = first;
807  first = c;
808  }
809  else if(lookup[c] > lookup[second])
810  {
811  second = c;
812  }
813  }
814  if(lookup[second] == 0)
815  {
816  second = first;
817  }
818 
819  targetPtr.set(first << pixelOffset | second);
820  }
821  }
822 }

838 Scroom::Utils::Stuff Operations1bppClipped::cacheZoom(const ConstTile::Ptr tile, int zoom, Scroom::Utils::Stuff cache)
839 {
840  UNUSED(cache);
841 
842  if(zoom >= 0)
843  {
844  zoom = 0;
845  }
846 
847  const int pixelSize = 1 << (-zoom);
848  const int outputWidth = tile->width / pixelSize;
849  const int outputHeight = tile->height / pixelSize;
850 
851  const int stride = cairo_format_stride_for_width(CAIRO_FORMAT_ARGB32, outputWidth);
852  boost::shared_ptr<unsigned char> data = shared_malloc(stride * outputHeight);
853  Colormap::Ptr colormap = colormapProvider->getColormap();
854 
855  unsigned char* row = data.get();
856  for(int j = 0; j < outputHeight; j++, row += stride)
857  {
858  auto* pixel = reinterpret_cast<uint32_t*>(row);
859  for(int i = 0; i < outputWidth; i++)
860  {
861  int sum = 0;
862 
863  for(int y = 0; y < pixelSize; y++)
864  {
865  const byte* inputByte = tile->data.get() + (j * pixelSize + y) * tile->width / 8 + pixelSize * i / 8;
866  byte inputBit = pixelSize * i % 8;
867 
868  SampleIterator<const byte> bit(inputByte, inputBit);
869 
870  for(int x = 0; x < pixelSize; x++, ++bit)
871  {
872  if(*bit)
873  {
874  sum++;
875  }
876  }
877  }
878 
879  if(sum > 0)
880  {
881  sum = 1;
882  }
883  *pixel = colormap->colors[sum].getARGB32();
884  pixel++;
885  }
886  }
887 
888  return BitmapSurface::create(outputWidth, outputHeight, CAIRO_FORMAT_ARGB32, stride, data);
889 }

891 void Operations1bppClipped::reduce(Tile::Ptr target, const ConstTile::Ptr source, int x, int y)
892 {
893  // Reducing by a factor 8. Source tile is 1bpp. Target tile is 1bpp
894  int sourceStride = source->width / 8;
895  const byte* sourceBase = source->data.get();
896 
897  int targetStride = target->width / 8;
898  byte* targetBase = target->data.get() + target->height * y * targetStride / 8 + target->width * x / 8 / 8;
899 
900  for(int j = 0; j < source->height / 8; j++, targetBase += targetStride, sourceBase += sourceStride * 8)
901  {
902  // Iterate vertically over target
903  const byte* sourcePtr = sourceBase;
904  SampleIterator<byte> targetPtr(targetBase, 0);
905 
906  for(int i = 0; i < source->width / 8; i++, sourcePtr++, targetPtr++)
907  {
908  // Iterate horizontally over target
909 
910  // Goal is to compute a 8-bit grey value from a 8*8 black/white
911  // image. To do so, we take each of the 8 bytes, count the
912  // number of 1's in each, and add them. Finally, we divide that
913  // by 64 (the maximum number of ones in that area
914 
915  const byte* current = sourcePtr;
916  int sum = 0;
917  for(int k = 0; k < 8; k++, current += sourceStride)
918  {
919  sum += bcl.lookup(*current);
920  }
921 
922  targetPtr.set((sum > 0) ? 1 : 0);
923  }
924  }
